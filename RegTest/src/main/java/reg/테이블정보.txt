--회원가입시 취미정보 등록하기  (회원테이블, 취미테이블에 데이터 등록, 회원테이블이 선행되어야 한다!!) , 복잡한 형태 등록 경험하기
--(트랜잭션경험하기) 고객정보가 입력되지 않았는데 고객의 취미가 입력되면 안된다--
--모두 수행되거나 수행되지 않게 하기

drop table member2;
drop table  member_hobby2;
  


CREATE TABLE member2 (
    member_id NUMBER PRIMARY KEY,
    username  VARCHAR2(50),
    password  VARCHAR2(100),
    gender    CHAR(1),
    role2      VARCHAR2(20)
);

CREATE TABLE hobby2 (
    hobby_id NUMBER PRIMARY KEY,
    hobby_name VARCHAR2(50)
);

CREATE TABLE member_hobby2 (
    member_id NUMBER,
    hobby_id NUMBER,
    CONSTRAINT fk_member FOREIGN KEY(member_id) REFERENCES member2(member_id),
    CONSTRAINT fk_hobby FOREIGN KEY(hobby_id) REFERENCES hobby2(hobby_id),
    PRIMARY KEY (member_id, hobby_id)
);
